from datetime import date

class Huesped:
    def __init__(self, id_huesped, nombre, apellido, rut, telefono, correo):
        self.id = id_huesped
        self.nombre = nombre
        self.apellido = apellido
        self.rut = rut
        self.telefono = telefono
        self.correo = correo
    
    def __str__(self):
        return f"ID: {self.id} | Nombre: {self.nombre} {self.apellido} | RUT: {self.rut}"
